async function handleAuth() { let nameInput = document.getElementById("usernameInput").value.trim(); if (!nameInput) { alert("Введите никнейм!"); return; } let token = getDeviceToken(); try { // ИСПРАВЛЕНО: Автоматически берет адрес текущего сайта let response = await fetch(`${window.location.protocol}//${window.location.host}/auth`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username: nameInput, token: token }) }); let result = await response.json(); if (result.success) { myName = nameInput; document.getElementById("login-screen").style.display = "none"; document.getElementById("chat-screen").style.display = "flex"; connectWebSocket(); } else { alert(result.error); } } catch (error) { alert("Не удалось связаться с сервером."); } } function connectWebSocket() { // ИСПРАВЛЕНО: Автоматически переключается на wss:// в интернете let wsProtocol = window.location.protocol === "https:" ? "wss:" : "ws:"; ws = new WebSocket(`${wsProtocol}//${window.location.host}/ws/${myName}`); ws.onmessage = function(event) { let data = JSON.parse(event.data); if (data.type === "users") { allUsersData = data.users; updateUsersList(); } else if (data.type === "msg") { handleIncomingMessage(data.sender, data.text); } }; ws.onclose = function(event) { if (event.code === 4000) { alert("Вас выбили из сети!"); location.reload(); } }; }
